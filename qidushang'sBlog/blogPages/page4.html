Vue的v-for,v-on及v-if指令

列表展示(list)

<li v-for="item in movies">{{item}}</li>

data:{ movies:['内容','内容','内容'] }

计数器

v-on指令:监听事件

<h2>当前计数：{{counter}}</h2>

<button v-on:click="add">+</button>

data:{ counter:0 },
methods:{ add:function(){ console.log('add被执行'); this.counter++ },
sub:function(){ console.log('sub被执行'); this.counter-- } }

v-if指令:v-if="seen"

data:{ seen:true/false }
